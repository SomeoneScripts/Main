	function LoadSettings()
	    if isfile and readfile and isfile(SaveFile) then
	        local SuccessRead, DataEncrypted = pcall(readfile, SaveFile)

	                    S[K] = V
	                end
	            end
	        else
	            SS()
	        end
	    else
	        SS()
	    end

	function Format(s)
	    return s:gsub("(%S)(%S*)", function(a,b) return a:upper()..b end)
	end
	
	function Translate(text)
	    local textLower = text:lower()
	    local final = text
	    if Translations and S and S["Select The Language"] and S["Translate"] then
	        local lang = Translations[S["Select The Language"]]
	        if lang and lang.Universal and lang.Universal[textLower] then
	            final = lang.Universal[textLower]
	        else
	            local gameId = tostring(g.GameId)
	            local langTbl = lang and lang[gameId]
	            if langTbl and langTbl[textLower] then
	                final = langTbl[textLower]
	            end
	        end
	    end
	    return Format(final)
	end

	function Tween(Pos, Key)
	    Key = Key or "Cancel"
	    if OldTarget ~= Pos then
	        OldTarget = Pos
	    end
	    HRP.CanCollide = false
	    if not getgenv().NoSeat and Humanoid and Humanoid.Sit then

	    end
	    getgenv()[Key.."_Active"] = getgenv()[Key]

	        HRP.CFrame = CFrame.new(Pos)
	        HRP.AssemblyLinearVelocity = Vector3.zero
	        getgenv()[Key.."_Active"] = false
	        return
	    end
	    if math.abs(HRP.Position.Y - Pos.Y) > 50 then
	        HRP.CFrame = CFrame.new(HRP.Position.X, Pos.Y, HRP.Position.Z)
	    end
	    while getgenv()[Key] and GetDistance(Pos) > 5 do
	        if not getgenv().NoSeat and Humanoid and Humanoid.Sit then
	            Humanoid.Sit = false
	        end
	        if GetDistance(Pos) <= 200 then
	            HRP.CFrame = CFrame.new(Pos)
	            HRP.AssemblyLinearVelocity = Vector3.zero
	            getgenv()[Key.."_Active"] = false
				return
	        end
	        HRP.CFrame += (Pos - HRP.Position).Unit * 325 * task.wait()
	    end
	end

	function Hop()
	    local Path = "Vortex Hub/Servers/Blox Fruits"
	    local File = Path.."/Servers.json"
	    if not isfolder("Vortex Hub") then makefolder("Vortex Hub") end
	    if not isfolder("Vortex Hub/Main") then makefolder("Vortex Hub/Main") end
	    if not isfolder(Path) then makefolder(Path) end
	    if not isfile(File) then writefile(File, "{}") end
	    local VisitedServers = HttpService:JSONDecode(readfile(File))
	    local Cursor = ""
	    local CurrentHour = os.date("!*t").hour
	    function SaveServers()
	        writefile(File, HttpService:JSONEncode(VisitedServers))
	    end
	    function TPReturner()
	        local ServerData
	        if Cursor == "" then
	            ServerData = HttpService:JSONDecode(g:HttpGet("https://games.roblox.com/v1/games/"..PlaceId.."/servers/Public?sortOrder=Asc&limit=100"))
	        else
	            ServerData = HttpService:JSONDecode(g:HttpGet("https://games.roblox.com/v1/games/"..PlaceId.."/servers/Public?sortOrder=Asc&limit=100&cursor="..Cursor))
	        end
	        if ServerData.nextPageCursor and ServerData.nextPageCursor ~= "null" then
	            Cursor = ServerData.nextPageCursor
	        end
	        for _, Info in pairs(ServerData.data) do
	            local Id = tostring(Info.id)
	            if tonumber(Info.playing) < tonumber(Info.maxPlayers) and not table.find(VisitedServers, Id) then
	                table.insert(VisitedServers, Id)
	                SaveServers()
	                task.wait()
	                pcall(function()
	                    task.wait()
	                    TeleportService:TeleportToPlaceInstance(PlaceId, Id, LocalPlayer)
	                end)
	                task.wait()
	            end
	        end
	    end
	    function StartHop()
	        while task.wait() do
	            pcall(function()
	                TPReturner()
	                if Cursor ~= "" then TPReturner() end
	            end)
	        end
	    end
	    StartHop()
	end

	function BringMobs()
		task.wait(0.5)
		if getgenv().BringEnemies and bringmobs then
			local kids = Enemies:GetChildren()
			local n = #kids
			for i = 1, n do
				local Mob = kids[i]
				if Mob.Name == MobName then
					local Hum = Mob:FindFirstChildOfClass("Humanoid")
					local HRP = Mob:FindFirstChild("HumanoidRootPart")
					if Hum and Hum.Health > 0 and HRP then
						local dir = MobPos - HRP.Position
						dir = Vector3.new(dir.X,0,dir.Z)
						local dist = dir.Magnitude
						sethiddenproperty(LocalPlayer, "SimulationRadius", math.huge)
						if dist > 15 then
							HRP.Velocity = dir.Unit * math.clamp(dist*5,400,800) + Vector3.new(0,HRP.Velocity.Y,0)
						elseif dist > 1 then
							HRP.Velocity = dir.Unit * 150 + Vector3.new(0,HRP.Velocity.Y,0)
						else
							HRP.Velocity = Vector3.new(0,HRP.Velocity.Y,0)
						end
					end
				end
			end
		end
	end
